select I.ID, I.ISSUESLIST_ID, L.NAME, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO
from ISSUESLISTS L
join ISSUES I on L.ID = I.ISSUESLIST_ID;

select I.ID, I.ISSUESLIST_ID, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO, U.FIRSTNAME, U.LASTNAME
from USERS U
join ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

select U.ID, U.FIRSTNAME, U.LASTNAME, COUNT(*) as ALL_TASKS
from USERS U join ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
group by I.USER_ID_ASSIGNEDTO;
